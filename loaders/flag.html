# Flag

# Overview

Loads configuration from CLI flags.

# Usage

import * as Konfig from "@willsoto/node-konfig-core";

const loader = new Konfig.FlagLoader({
  cliOptions: {
    flags: {
      environment: {
        alias: "env",
        default: "development",
      },
      appName: {
        default: "default",
      },
    },
  },
});
